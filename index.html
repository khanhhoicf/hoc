---------burble-sort
 void swap(int &x, int &y){
    int temp = x;
    x = y;
    y = temp;
}
            
void bubbleSort(int a[], int n)
                {
                 for (int i = 0; i < n - 1; i++)
                     for (int j = n - 1; j > i; j--)
                            if (a[j] < a[j - 1])
                            swap(a[j], a[j - 1]);
                }

---------Selection-sort
void Selection(int a[], int n)
            {
                for (int i = 0; i < n - 1; i++)
                {
                    int min = i;
                    for (int j = i + 1; j < n; j++)
                    {
                        if (a[min] > a[j])
                            min = j;
                    }
                    if (i != min)
                        swap(a[i], a[min]);
                }
            }
